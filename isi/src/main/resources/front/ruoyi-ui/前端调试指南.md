### 1. 验证码获取流程 (login.vue)

```javascript
getCode() {
  console.log('====== 开始获取验证码 ======')
  getCodeImg().then(res => {
    console.log('axios 响应拦截器返回的数据:', res)
    const data = res.data || res
    console.log('解析后的 data 对象:', data)
    
    this.captchaEnabled = data.captchaEnabled === undefined ? true : data.captchaEnabled
    console.log('captchaEnabled:', this.captchaEnabled)
    
    if (this.captchaEnabled) {
      this.codeUrl = "data:image/gif;base64," + data.img
      this.loginForm.uuid = data.uuid
      console.log('验证码图片 URL:', this.codeUrl.substring(0, 50) + '...')
      console.log('验证码 UUID:', this.loginForm.uuid)
    }
    console.log('====== 验证码获取完成 ======')
  }).catch(error => {
    console.error('获取验证码失败:', error)
    console.error('错误详情:', error.response || error.message)
  })
}
```

### 2. 登录流程 (store/modules/user.js)

```javascript
Login({ commit }, userInfo) {
  console.log('====== 开始登录 ======')
  console.log('登录信息:', { username, code, uuid })
  
  return new Promise((resolve, reject) => {
    login(username, password, code, uuid).then(res => {
      console.log('登录响应:', res)
      const token = res.token || res.data?.token
      console.log('提取的 token:', token)
      
      if (token) {
        setToken(token)
        commit('SET_TOKEN', token)
        console.log('Token 已保存')
        resolve()
      } else {
        console.error('未找到 token，响应数据:', res)
        reject(new Error('登录失败：未获取到 token'))
      }
    }).catch(error => {
      console.error('登录请求失败:', error)
      reject(error)
    })
  })
}
```

## 常见错误排查

### ❌ 问题 1：只显示"获取验证码"，不显示后续日志

**可能原因：**
- axios 请求被 CORS 策略阻止
- 后端服务未启动
- 请求 URL 配置错误

**检查方法：**
1. 查看 Network 标签，看请求是否发出
2. 检查请求的 Status 是什么
3. 查看是否有 CORS 相关错误

**解决方案：**
```bash
# 确保后端服务已启动
cd D:\study_project\lingxi\isi
# 运行主程序
```

### ❌ 问题 2：显示"获取验证码失败"

**可能原因：**
- 后端接口报错
- 网络问题
- 跨域配置问题

**检查方法：**
```javascript
// 查看详细的错误信息
console.error('错误详情:', error.response || error.message)
```

**常见错误及解决：**

| 错误信息 | 原因 | 解决方法 |
|---------|------|---------|
| Network Error | 后端服务未启动 | 启动后端服务 |
| timeout | 请求超时 | 检查后端是否卡住 |
| 404 Not Found | 接口路径错误 | 检查 @RequestMapping 配置 |
| 500 Internal Server Error | 后端代码报错 | 查看后端日志 |

### ❌ 问题 3：登录时显示"未找到 token"

**可能原因：**
- 后端返回的数据结构不对
- 前端解析逻辑有误

**检查方法：**
```javascript
// 查看完整的响应数据
console.error('未找到 token，响应数据:', res)
```
